<!DOCTYPE html>
<html>
  <head>
    <title> WebSocket Test Page </title>

    <script>
      function tryme()
      {
        if ( window.WebSocket )
        {
          alert( "YES! WebSocket is available in this browser." );
        }
        else
        {
          alert( "YES! WebSocket is NOT available in this (sucky) browser." );
        }
        // also use F12 console in browser and enter window.WebSocket
      }

      var log = function( s )
      {
        if ( document.readyState !== 'complete' )
        {
          log.buffer.push( s );
        }
        else
        {
          document.getElementById( 'output' ).textContent += ( s + "\n" );
        }
      }

      log.buffer = [];

      if ( this.MozWebSocket )
      {
        WebSocket = MozWebSocket;
      }

      url = 'ws://localhost:8998/echo';
      w = new WebSocket( url );
      w.onopen = function()
      {
        log( 'open' );
        w.binaryType = 'arraybuffer';
        w.send( 'test message for Web Socket request' );
        var a = new Uint8Array( [ 1,2,3,4,5,6,7,8 ] );
        w.send( a.buffer );
      }
      w.onmessage = function( e )
      {
        log( e.data.toString() );
      }
      w.onclose = function( e )
      {
        log( 'closed' );
      }
      w.onerror = function( e )
      {
        log( 'error' );
      }

      window.onload = function()
      {
        log( log.buffer.join("\n") );
        document.getElementById( 'sendButton' ).onclick = function()
        {
            w.send( document.getElementById( 'inputMessage' ).value );
        }
      }
    </script>
  </head>

  <body onload="tryme();">
    <input type="text" id="inputMessage" value="Bieberlicious!"><button id="sendButton">Send</button>
    <pre id="output"></pre>
  </body>
