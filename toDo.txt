	// Set up Hex to RGB Conversion
	function hexToR(h) {
		if(h.length == 3){
		  h = h + h;
		}
		return parseInt((cutHex(h)).substring(0,2),16)
	}
	
	function hexToG(h) {
		if(h.length == 3){
		 h = h + h;
		}
		return parseInt((cutHex(h)).substring(2,4),16)
	}
	
	function hexToB(h) {
		if(h.length == 3){
		 h = h + h;
		}
		return parseInt((cutHex(h)).substring(4,6),16)
	}
	
	// Find colors compliment
	function findCompliment(hex){
	
		thisrgb = new Object();
		thisrgb.r= hexToR(hex);
		thisrgb.g= hexToR(hex);
		thisrgb.b= hexToB(hex);
		var temprgb=thisrgb;
		temphsv=RGB2HSV(temprgb);
		temphsv.hue=HueShift(temphsv.hue,180.0);
		temprgb=HSV2RGB(temphsv);
		return "rgb(" + temprgb.r +"," + temprgb.g +","+temprgb.b+")";
